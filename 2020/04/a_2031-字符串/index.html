思考与坑点

本题借鉴呆神思路

这题是关于字符串的，字符串本身比较头疼，这题还行，主要找到n1，n2，n3和空格的关系。这题可以一行一行输出，因为都是有规律的，不像螺旋矩阵。

  • n1==n3
  • n2=(len-2*n1)+2

  • 空格数=n2-2 –> 空格数=len-2*n1

    AC代码

    #include<iostream>
#include<stdlib.h>
#include<stdio.h>
#include<string.h>
#include<math.h>
#include<algorithm>
#include<vector>
#include<string>
#pragma warning(disable:4996)
using namespace std;

int main() 
{
#ifdef ONLINE_JUDGE
#else
	freopen("1.txt", "r", stdin);
#endif
	
	int n1;
	string str,temp;

	cin >> str;
	int len = str.size();
	
	n1 = (len + 2) / 3;

	int k = len - (2 * n1);

	for (int i = 0; i < k; i++)
		temp.push_back(' ');
	
	int i;
	for (i = 0; i < n1-1; i++)
	{
		cout << str[i];
		cout << temp;
		cout << str[len - 1 - i] << endl;;
	}

	//这地方你用substr也是可以的，只要知道n2是几个字符就行
	for (int j = i; j < i+k + 2; j++)
	{
		cout << str[j];
	}
	return 0;
}
